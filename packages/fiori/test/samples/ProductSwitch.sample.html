<script>
	var shellBar = document.getElementById("shellbar");
	var popover = document.getElementById("productswitch-popover");

	shellbar.addEventListener("product-switch-click", event => {
		if (popover.opened) {
			popover.close();
		} else {
			popover.showAt(event.detail.targetRef);
		}
	});
</script>
