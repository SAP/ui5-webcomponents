import { Story, Preview, ArgsTable } from "@storybook/addon-docs";

import argTypes from "./argsTypes.js";
import Description from "./Description.md";

<Meta title="Components/Select" component="ui5-select" argTypes={argTypes} />

export const Template = (args) => {
  return html`
    <ui5-select
      accessible-name="${ifDefined(args.accessibleName)}"
      accessible-name-ref="${ifDefined(args.accessibleNameRef)}"
      ?disabled="${ifDefined(args.disabled)}"
      ?required="${ifDefined(args.required)}"
      name="${ifDefined(args.name)}"
      selected-option="${ifDefined(args.selectedOption)}"
      value-state="${ifDefined(args.valueState)}"
      value-state-message="${ifDefined(args.valueStateMessage)}"
      @ui5-change="${ifDefined(args["ui5-change"])}"
    >
      ${args.defaultSlot}
    </ui5-select>
  `;
};
