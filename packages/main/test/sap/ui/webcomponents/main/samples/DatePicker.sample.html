Basic DatePicker


<ui5-datepicker id="myDatepicker1"></ui5-datepicker>
<script>
	import { getFormatterInstance } from "@ui5/webcomponents-base/src/sap/ui/webcomponents/base/dates/DateFormat";
	getFormatterInstance({
		style: "short"
	}).then(formatter => {
		const dp = document.getElementById('myDatepicker1');
		dp.setAttribute('value', formatter.format(new Date()));
	});
</script>

DatePicker with Placeholder, Tooltip, Events, and ValueState


<ui5-datepicker id="myDatepicker2" placeholder='Delivery Date...' title='Delivery Date!'></ui5-datepicker>
<script>
	const dp = document.getElementById('myDatepicker2');
	dp.addEventListener('change', (e) => {
		dp.setAttribute('value-state', e.detail.valid ? 'None' : 'Error');
		console.log('change: ', e.detail);
	});
	dp.addEventListener('liveChange', (e) => console.log('liveChange: ', e.detail));
</script>
